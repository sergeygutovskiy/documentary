Описание

Вывести данные врачей с информацией об их рабочих днях

SQL код

select name, class, working_days from doctors
inner join schedule on doctors.schedule_id = schedule.id
order by name desc


2 запрос

Описание

Вывести данные о приёмах с отображением информации о пациентах

SQL код

select * from visits
inner join patients on visits.patient_id = patients.id
order by was_at

9 запрос

Описание

Вывести имена доктора и пациента, который к нему приходил, а также дату приёма

SQL код

select doctors.name, patients.name, visits.was_at from visits
inner join doctors on (visits.doctor_id = doctors.id)
inner join patients on (visits.patient_id = patients.id)


10 запрос

Описание

Вывести данные о докторах с количеством пациентов, приходивших к ним на приём

SQL код

select doctors.id, doctors.name, count(*) from visits
inner join doctors on (visits.doctor_id = doctors.id)
inner join patients on (visits.patient_id = patients.id)
group by doctors.id


11 запрос

Описание

Вывести данные о заработанных деньгах каждого врача от проведённых услуг

SQL код

select visits.doctor_id, sum(visits.price) as sum from visits
inner join doctors on (visits.doctor_id = doctors.id)
group by visits.doctor_id


12 запрос

Описание

Вывести дату(ы) приёмов, в которую(ые) было от 2-х до 3-х приёмов пациентов

SQL код

select was_at, count(*) from visits
group by date(was_at)
having count(*) in (2,3)
